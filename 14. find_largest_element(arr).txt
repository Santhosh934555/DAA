def find_largest_element(arr):
    if len(arr) == 0:
        return None
    else:
        return max(arr)

# Example array
my_array = [10, 5, 20, 8, 15]

# Find the largest element
largest_element = find_largest_element(my_array)
print(f"The largest element in the array is: {largest_element}")
